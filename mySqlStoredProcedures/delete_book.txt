CREATE DEFINER=`root`@`localhost` PROCEDURE `delete_book`(
bookID INT,
bookStatus TEXT
)
BEGIN
	DELETE books, media_pieces, media_genre
    FROM books
    
    JOIN media_pieces
    ON books.book_id = media_pieces.media_id AND media_pieces.media_type_id = 1
    
    JOIN media_genre
    ON media_pieces.media_id = media_genre.media_id AND media_genre.media_type_id = 1
    
    WHERE books.book_id = bookID AND media_pieces.status = bookStatus;
END
