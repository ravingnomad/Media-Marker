CREATE DEFINER=`root`@`localhost` PROCEDURE `update_book`(
book_id INT,
fieldToChange TEXT,
newValue TEXT
)
BEGIN
	IF LOWER(fieldToChange) = 'title' THEN
		UPDATE books
		SET books.title = newValue
		WHERE books.book_id = book_id;
	ELSEIF LOWER(fieldToChange) = 'author' THEN
		UPDATE books
		SET books.author = newValue
		WHERE books.book_id = book_id;
    END IF;
END
